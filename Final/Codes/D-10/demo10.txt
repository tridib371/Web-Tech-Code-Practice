            $uname = filter_var($_POST["uname"], FILTER_SANITIZE_EMAIL);
            if (filter_var($uname, FILTER_VALIDATE_EMAIL)) {
                  $pass = $_POST["pass"];
                  $conn = mysqli_connect('localhost', 'root', '', 'app_users');
                  $sql = "INSERT INTO records(username, password) VALUES('$uname', '$pass')";
